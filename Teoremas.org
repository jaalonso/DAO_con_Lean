# Resumen de teoremas

Los teoremas utilizados son los siguientes:

#+begin_src lean
+ add_assoc : (a + b) + c = a + (b + c) 
+ add_comm  : a + b = b + a 
+ add_mul   : (a + b) * c = a * c + b * c 
+ add_sub   : a + (b - c) = (a + b) - c
+ add_zero  : a + 0 = a
+ and.left  : P ∧ Q → P
+ eq.trans  : a = b → b = c → a = c
+ id        : P → P
+ mul_assoc : (a * b) * c = a * (b * c)   
+ mul_comm  : a * b = b * a 
+ mul_sub   : a * (b - c) = a * b - a * c 
+ pow_two   : a^2 = a * a
+ sub_self  : a - a = 0
+ sub_sub   : (a - b) - c = a - (b + c)
+ two_mul   : 2 * a = a + a
#+end_src
